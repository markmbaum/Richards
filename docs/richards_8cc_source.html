161 void Richards::update_q (double *w, double t) {
162 
163  //index
164  long i;
165  //infiltration flag
166  bool infil = f_infil(t);
167 
168  //----------------------------------------------
169  //cell edge saturations and saturation gradients
170 
171  //bottom
172  we[0] = f_w_bot(poroe[0]);
173  dwdz[0] = (w[0] - we[0])/(delz[0]/2);
174  //interior
175  for (i=1; i<n; i++) {
176  we[i] = vefac[i]*w[i] + (1.0 - vefac[i])*w[i-1];
177  dwdz[i] = gefac[i]*(w[i] - w[i-1]);
178  }
179  //top value depends on infiltration flag
180  infil ? we[n] = poroe[n] : NAN;
181  dwdz[n] = (we[n] - w[n-1])/(delz[n-1]/2);
182 
183  //------------------------------
184  //cell edge hydraulic properties
185 
186  for (i=0; i<n+1; i++) {
187  K[i] = f_K(we[i], Ksat[i], poroe[i], stg.b);
188  dpsidw[i] = f_dpsidw(we[i], psisat[i], poroe[i], stg.b);
189  D[i] = K[i]*dpsidw[i];
190  }
191 
192  //------
193  //fluxes
194 
195  //water table
196  q[0] = f_q(K[0], dpsidw[0], dwdz[0], stg.wilt, w[0]/poroc[0]);
197  //cell-to-cell exchange
198  for (i=1; i<n; i++)
199  q[i] = f_q(K[i], dpsidw[i], dwdz[i], w[i-1]/poroc[i-1], w[i]/poroc[i-1]);
200  //surface infiltration/evaporation
201  if ( infil ) {
202  q[n] = f_q(K[n], dpsidw[n], dwdz[n], w[n-1]/poroc[n-1], stg.wilt);
203  } else {
204  q[n] = stg.Levap*(w[n-1] - stg.wilt*poroe[n])/stg.tauevap;
205  }
206 }

219 double Richards::dt_adapt () {
220 
221  //compute fraction of maximum stable time step
222  double dt = INFINITY;
223  double dtmax;
224  for (long i=0; i<n+1; i++) {
225  dtmax = dtcons[i]/D[i];
226  if ( (dtmax < dt) && !std::isnan(dtmax) )
227  dt = dtmax;
228  }
229  dt *= stg.dtfac;
230  //infiltration management
231  double t, ta, tb;
232  t = get_sol(n);
233  infil_times(get_sol(n), &ta, &tb);
234  if ( (t >= ta) && (t <= tb) ) {
235  if ( dt > stg.infdur/1000 )
236  dt = stg.infdur/1000;
237  if ( t + dt > tb )
238  dt = tb - t;
239  } else {
240  if ( t + dt > ta )
241  dt = ta - t;
242  }
243  //evaporation management
244  if ( dt > stg.tauevap*delz[0]/stg.Levap )
245  dt = 0.99*stg.tauevap*delz[0]/stg.Levap;
246 
247  return( dt );
248 }

250 void Richards::steady (double atol, unsigned long ntol) {
251 
252  //index
253  long i;
254  //step counter
255  unsigned long count = 0;
256  //euler time step
257  double dt = INFINITY;
258  //alias the solution array
259  double *w = get_sol();
260  //storage for time derivatives, initially at inf
261  double *dwdt = new double[n+1];
262  for (i=0; i<n; i++) dwdt[i] = INFINITY;
263 
264  //integrate until dwdt is small enough or too many steps are taken
265  while ( (absmax(dwdt, n)*dt > atol) && (count < ntol) ) {
266  ode_fun(w, dwdt);
267  dt = 0.75*dt_adapt();
268  for (i=0; i<n; i++)
269  w[i] += dt*dwdt[i];
270  count++;
271  }
272 
273  delete [] dwdt;
274 }

276 void Richards::spinup (double rtol, bool quiet) {
277 
278  long i;
279  if ( !quiet ) {
280  printf(" max rel dif requirement is %g\n", rtol);
281  printf(" PERIODS | MAX REL DIF \n");
282  printf(" ------- | -------------\n");
283  }
284  //integrate over a single infiltration period to get started
285  solve_adaptive(stg.infper, stg.infper/1e12, false);
286  //store the bottom boundary flux
287  std::vector<double> q_a(n, INFINITY);
288  std::vector<double> q_b;
289  for (i=0; i<n; i++) q_b.push_back(q[i]);
290  //maximum relative difference
291  double mrd = maxreldif(q_a, q_b, n);
292  //continue integrating over infiltration periods until the qb is stable
293  long count = 0;
294  long ord = floor(log10(mrd));
295  while ( (mrd > rtol) || (count <= 5) ) {
296  solve_adaptive(stg.infper, stg.infper/1e12, false);
297  count++;
298  for (i=0; i<n; i++) {
299  q_a[i] = q_b[i];
300  q_b[i] = q[i];
301  }
302  mrd = maxreldif(q_a, q_b, n);
303  if ( (!quiet) && (floor(log10(mrd)) != ord ) ) {
304  printf(" %-7li | %-11g\n", count, mrd);
305  ord = floor(log10(mrd));
306  }
307  }
